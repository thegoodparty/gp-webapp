import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import Config from '../tailwind.config.js';
import { pick } from 'es-toolkit';

export function renderColorSwatches(...colorKeys) {
  return Object.entries(pick(Config.theme.extend.colors, colorKeys)).map(
    ([colorName, colorValues]) => {
      const colorTitle = `${colorName[0].toUpperCase()}${colorName.slice(1)}`;
      return (
        <ColorItem key={colorName} title={colorTitle} colors={colorValues} />
      );
    },
  );
}

<Meta title="Theme/Colors" />
<ColorPalette>
  {renderColorSwatches(
    'gray',
    'red',
    'green',
    'blue',
    'orange',
    'indigo',
    'lime',
    'purple',
    'mint',
    'cyan',
  )}
</ColorPalette>
